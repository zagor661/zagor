-- Uruchom w Supabase → SQL Editor
-- Tabela subskrypcji push notifications

CREATE TABLE IF NOT EXISTS push_subscriptions (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  profile_id UUID NOT NULL REFERENCES profiles(id) ON DELETE CASCADE,
  location_id UUID NOT NULL,
  endpoint TEXT NOT NULL,
  p256dh TEXT NOT NULL,
  auth TEXT NOT NULL,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(profile_id, endpoint)
);

CREATE INDEX idx_push_location ON push_subscriptions(location_id);
CREATE INDEX idx_push_profile ON push_subscriptions(profile_id);

-- RLS
ALTER TABLE push_subscriptions ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Anyone can insert push sub" ON push_subscriptions FOR INSERT WITH CHECK (true);
CREATE POLICY "Anyone can read push subs" ON push_subscriptions FOR SELECT USING (true);
CREATE POLICY "Anyone can delete own push sub" ON push_subscriptions FOR DELETE USING (true);
